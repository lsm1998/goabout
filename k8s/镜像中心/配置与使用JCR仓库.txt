1.登录 JCR 并创建 5 个仓库
  分别创建： docker-local, docker-test, docker-release, docker-remote 和 docker virtual 仓库

2.上传镜像
docker pull registry.cn-beijing.aliyuncs.com/qingfeng666/kubeblog:1.0

docker tag registry.cn-beijing.aliyuncs.com/qingfeng666/kubeblog:1.0 art.local:8081/docker-local/kubeblog:1.0

docker push art.local:8081/docker-local/kubeblog:1.0



# nginx
docker pull nginx
docker tag nginx art.local:8081/docker-local/nginx:1.0
docker push art.local:8081/docker-local/nginx:1.0

# redis
docker pull redis
docker tag redis art.local:8081/docker-local/redis:1.0
docker push art.local:8081/docker-local/redis:1.0
