docker pull gitlab/gitlab-ce

mkdir -p /gitlab/config
mkdir -p /gitlab/logs
mkdir -p /gitlab/data

# 创建目录进行挂载
/server/gitlab/

# 启动镜像
docker run -d --publish 443:443 --publish 80:80 --publish 8022:22 --hostname 192.168.0.12 --name gitlab \
--restart always -v \
/gitlab/config:/etc/gitlab -v \
/gitlab/logs:/var/log/gitlab -v \
/gitlab/data:/var/opt/gitlab -v \
/etc/localtime:/etc/localtime:ro --privileged=true gitlab/gitlab-ce
